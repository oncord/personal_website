Express.js API Endpoint


const express = require('express');
const mongoose = require('mongoose');
const Temperature = require('./models/Temperature');

const router = express.Router();

// GET temperature data
router.get('/api/temperature', async (req, res) => {
    try {
        const temperatures = await Temperature
            .find()
            .sort({ timestamp: -1 })
            .limit(100);
        
        res.json(temperatures);
    } catch (error) {
        res.status(500).json({ error: 'Failed to fetch temperature data' });
    }
});

// POST new temperature reading
router.post('/api/temperature', async (req, res) => {
    try {
        const { value, location, timestamp } = req.body;
        const temperature = new Temperature({
            value,
            location,
            timestamp: timestamp || new Date()
        });
        
        await temperature.save();
        res.status(201).json(temperature);
    } catch (error) {
        res.status(400).json({ error: 'Invalid temperature data' });
    }
});

module.exports = router;
